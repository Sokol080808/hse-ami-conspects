\section{Информация}

\subsection{Оценка}
Round\( (0.275 \cdot O_{HW} + 0.625 \cdot O_{PRJ} + 0.1 \cdot O_{SEM}) \)

\subsection{Ревью}
\begin{itemize}
    \item Ручная проверка кода семинаристами или ассисетентами
    \item Засчитывается время последнего коммита
    \item Максимальное время ожидания ревью --- 72 часа
    \item Защита некоторых задач с ассистентом очно или онлайн
\end{itemize}

\subsection{Дедлайны по ДЗ}
\begin{itemize}
    \item Две недели на сдачу
    \item До дедлайна \( 100 \)\% 
    \item После дедлайна оценка линейно убывает до \( 0 \)\% через неделю
\end{itemize}

\subsection{Проект}
\begin{itemize}
    \item Законченная программа
    \item Сильно больше кода, чем в ДЗ
    \item Обязательное ревью
\end{itemize}
